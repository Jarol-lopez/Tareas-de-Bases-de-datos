--5. Mediante dos consultas recuperar el salario máximo y el salario mínimo de la
--empresa e indicar su diferencia 

DECLARE  
SALARIO_MAXIMO number(10);
SALARIO_MINIMO number(10);
    
BEGIN
SELECT MAX(SALARY) INTO SALARIO_MAXIMO FROM EMPLOYEES;
SELECT MIN(SALARY) INTO SALARIO_MINIMO FROM EMPLOYEES;
DBMS_OUTPUT.PUT_LINE('EL SALARIO MAXIMO ES ');
DBMS_OUTPUT.PUT_LINE(SALARIO_MAXIMO);
DBMS_OUTPUT.PUT_LINE('EL SALARIO MINIMO ES ');
DBMS_OUTPUT.PUT_LINE(SALARIO_MINIMO);
DBMS_OUTPUT.PUT_LINE('LA DIFERENCIA ENTRE LO SALRIOS ES ');
DBMS_OUTPUT.PUT_LINE(SALARIO_MAXIMO - SALARIO_MINIMO);
    
 END;
